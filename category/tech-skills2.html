  1. 下载源码 
cd /usr/local/src
wget https://openresty.org/download/openresty-1.13.6.2.tar.gz
  1. 解压后编译安装 
yum install pcre-devel openssl-devel gcc curl  #需要事先准备好需要的依赖包
tar xzf openresty-1.13.6.2.tar.gz
cd openresty-1.13.6.2
